# Next.js Gateway Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Scaffold a Next.js 14 App Router gateway at `claude/gateway/` that proxies ClaimsApi and TaskApi, serves a live health dashboard, and ships as three git branches for three product tiers.

**Architecture:** A single Next.js codebase with shared proxy route handlers and a `lib/` layer; the dashboard page and health aggregator live on `main`; `gateway/proxy-only` strips them to a pure gateway; `gateway/portal` adds NextAuth.js, CRUD pages, and shadcn/ui. Health data is fetched by calling backend `GET /claims` and `GET /tasks` directly — neither API exposes a `/health` endpoint.

**Tech Stack:** Next.js 14 (App Router), TypeScript 5, Vitest 1 + React Testing Library 16 + msw 2 (unit/integration), Playwright 1.44 (E2E), Tailwind CSS 3, shadcn/ui (portal branch only), NextAuth.js 5 (portal branch only).

## Global Constraints

- Node.js ≥ 20; npm workspaces not used — gateway is a standalone package at `claude/gateway/`
- All work starts from `claude/` parent directory unless stated otherwise
- `CLAIMS_API_URL` and `TASK_API_URL` must be set; `lib/backend.ts` throws at module load if missing
- Proxy forwards only `authorization`, `x-api-key`, `content-type` — all other headers stripped
- Every upstream call uses `AbortController` with `TIMEOUT_MS = 10_000`; returns HTTP 504 on abort
- ClaimsApi fields: `claimId` (string/UUID), `status` (Submitted|UnderReview|Approved|Denied|Paid), `lastUpdated` (ISO string)
- TaskApi fields: `id` (string/UUID), `title` (string), `description` (string|null), `isComplete` (bool), `createdAt` (ISO string)
- `npm test` runs from `claude/gateway/`; all test paths are relative to that directory
- Commit after every task; branch names: `main`, `gateway/proxy-only`, `gateway/portal`

```ts
import { NextRequest, NextResponse } from 'next/server'

const CLAIMS_API_URL = process.env.CLAIMS_API_URL
const TASK_API_URL = process.env.TASK_API_URL

if (!CLAIMS_API_URL) throw new Error('CLAIMS_API_URL env var is not set')
if (!TASK_API_URL) throw new Error('TASK_API_URL env var is not set')

export const BACKEND_URLS = {
  claims: CLAIMS_API_URL,
  tasks: TASK_API_URL,
} as const

const FORWARDED_HEADERS = ['authorization', 'x-api-key', 'content-type']
export const TIMEOUT_MS = 10_000

export async function fetchWithTimeout(
  url: string,
  init: RequestInit = {},
  timeoutMs = TIMEOUT_MS
): Promise<Response> {
  const controller = new AbortController()
  const id = setTimeout(() => controller.abort(), timeoutMs)
  try {
    return await fetch(url, { ...init, signal: controller.signal })
  } finally {
    clearTimeout(id)
  }
}

export async function fetchUpstream(
  service: keyof typeof BACKEND_URLS,
  req: NextRequest,
  pathSegments: string[]
): Promise<NextResponse> {
  const base = BACKEND_URLS[service]
  const path = pathSegments.join('/')
  const url = `${base}/${path}${req.nextUrl.search}`

  const headers = new Headers()
  for (const key of FORWARDED_HEADERS) {
    const val = req.headers.get(key)
    if (val) headers.set(key, val)
  }

  let response: Response
  try {
    response = await fetchWithTimeout(url, {
      method: req.method,
      headers,
      body: req.method !== 'GET' && req.method !== 'HEAD' ? req.body : undefined,
      // @ts-expect-error — Node 18+ fetch needs duplex for streaming body
      duplex: 'half',
    })
  } catch {
    return NextResponse.json({ error: 'upstream timeout' }, { status: 504 })
  }

  const body = await response.arrayBuffer()
  return new NextResponse(body, {
    status: response.status,
    headers: {
      'content-type': response.headers.get('content-type') ?? 'application/json',
    },
  })
}
```

import { fetchWithTimeout, BACKEND_URLS } from '@/lib/backend'
import type { ClaimsHealth, TasksHealth, HealthResponse, Claim, Task } from '@/lib/types'

async function getClaimsHealth(): Promise<ClaimsHealth> {
  const start = Date.now()
  try {
    const res = await fetchWithTimeout(`${BACKEND_URLS.claims}/claims`)
    const latencyMs = Date.now() - start
    if (!res.ok) {
      return { status: 'degraded', latencyMs, totalClaims: 0, recentClaims: [] }
    }
    const claims: Claim[] = await res.json()
    return {
      status: 'up',
      latencyMs,
      totalClaims: claims.length,
      recentClaims: claims.slice(0, 5),
    }
  } catch {
    return { status: 'down', latencyMs: Date.now() - start, totalClaims: 0, recentClaims: [] }
  }
}

async function getTasksHealth(): Promise<TasksHealth> {
  const start = Date.now()
  try {
    const res = await fetchWithTimeout(`${BACKEND_URLS.tasks}/tasks`)
    const latencyMs = Date.now() - start
    if (!res.ok) {
      return { status: 'degraded', latencyMs, totalTasks: 0, recentTasks: [] }
    }
    const tasks: Task[] = await res.json()
    return {
      status: 'up',
      latencyMs,
      totalTasks: tasks.length,
      recentTasks: tasks.slice(0, 5),
    }
  } catch {
    return { status: 'down', latencyMs: Date.now() - start, totalTasks: 0, recentTasks: [] }
  }
}

export async function getHealthResponse(): Promise<HealthResponse> {
  const [claims, tasks] = await Promise.allSettled([getClaimsHealth(), getTasksHealth()])

  const DOWN_CLAIMS: ClaimsHealth = { status: 'down', latencyMs: 0, totalClaims: 0, recentClaims: [] }
  const DOWN_TASKS: TasksHealth = { status: 'down', latencyMs: 0, totalTasks: 0, recentTasks: [] }

  return {
    services: {
      claims: claims.status === 'fulfilled' ? claims.value : DOWN_CLAIMS,
      tasks: tasks.status === 'fulfilled' ? tasks.value : DOWN_TASKS,
    },
    generatedAt: new Date().toISOString(),
  }
}

import { BACKEND_URLS, fetchWithTimeout } from '@/lib/backend'
import type { Claim, Task } from '@/lib/types'

export interface ClaimWithTasks {
  claim: Claim
  tasks: Task[]
}

export async function getClaimWithTasks(claimId: string): Promise<ClaimWithTasks | null> {
  const [claimRes, tasksRes] = await Promise.all([
    fetchWithTimeout(`${BACKEND_URLS.claims}/claims/${claimId}`),
    fetchWithTimeout(`${BACKEND_URLS.tasks}/tasks`),
  ])
  if (!claimRes.ok) return null
  const claim: Claim = await claimRes.json()
  const allTasks: Task[] = tasksRes.ok ? await tasksRes.json() : []
  // Tasks linked by claimId convention — filter tasks whose description contains the claimId
  const tasks = allTasks.filter((t) => t.description?.includes(claimId) ?? false)
  return { claim, tasks }
}

'use client'
import { useState, useEffect } from 'react'
import { TaskCard } from '@/components/TaskCard'
import { CreateTaskForm } from '@/components/CreateTaskForm'
import type { Task } from '@/lib/types'

export default function TasksPage() {
  const [tasks, setTasks] = useState<Task[]>([])

  useEffect(() => {
    fetch('/api/tasks').then((r) => r.json()).then(setTasks).catch(() => {})
  }, [])

  async function handleComplete(id: string) {
    const res = await fetch(`/api/tasks/${id}/complete`, { method: 'POST' })
    if (res.ok) {
      const updated: Task = await res.json()
      setTasks((prev) => prev.map((t) => t.id === id ? updated : t))
    }
  }

  async function handleDelete(id: string) {
    const res = await fetch(`/api/tasks/${id}`, { method: 'DELETE' })
    if (res.ok) setTasks((prev) => prev.filter((t) => t.id !== id))
  }

  return (
    <main className="p-8 max-w-4xl mx-auto">
      <h1 className="text-2xl font-bold text-white mb-6">Tasks</h1>
      <div className="mb-8">
        <h2 className="text-lg font-semibold text-white mb-3">New Task</h2>
        <CreateTaskForm onCreated={(t) => setTasks((prev) => [t, ...prev])} />
      </div>
      <div className="grid gap-3">
        {tasks.map((t) => (
          <TaskCard key={t.id} task={t} onComplete={handleComplete} onDelete={handleDelete} />
        ))}
      </div>
    </main>
  )
}
